INCLUDE(UseSWIG)

SET(EXTRA_LINK_ARGS "${CMAKE_SHARED_LINKER_FLAGS}")
IF(APPLE)
  IF((CMAKE_CXX_COMPILER_ID MATCHES "GNU"  ) OR
     (CMAKE_CXX_COMPILER_ID MATCHES "Clang"))
    SET(EXTRA_LINK_ARGS "${EXTRA_LINK_ARGS} -undefined dynamic_lookup")
  ENDIF()
ENDIF()

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR})
INCLUDE_DIRECTORIES(${PYTHON_INCLUDE_PATH})

SET(HEADERS "SalinePy.hh")
SET(SOURCES "SalinePy.cc")

SET(SWIG_RUNTIME swigpyrun.h)
ADD_CUSTOM_COMMAND(
  OUTPUT ${SWIG_RUNTIME}
  COMMAND ${SWIG_EXECUTABLE} -python -external-runtime
)
LIST(APPEND HEADERS ${CMAKE_CURRENT_BINARY_DIR}/${SWIG_RUNTIME})

TRIBITS_ADD_LIBRARY(SalinePy
 SHARED
 HEADERS ${HEADERS}
 SOURCES ${SOURCES}
)

SET_TARGET_PROPERTIES(SalinePy
  PROPERTIES LINK_FLAGS "${EXTRA_LINK_ARGS}"
)
LIST(APPEND CMAKE_SWIG_FLAGS -py3)

SET(ADDITIONAL_CLEAN_FILES "")

SET_SOURCE_FILES_PROPERTIES(SalinePy.i PROPERTIES CPLUSPLUS ON)
SWIG_ADD_LIBRARY(SalinePy TYPE SHARED LANGUAGE python SOURCES SalinePy.i)
SWIG_LINK_LIBRARIES(SalinePy saline)
